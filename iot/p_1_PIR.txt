/* 
   === PIR Motion Sensor with Arduino ===
   Connections:
   - PIR sensor VCC  →  5V on Arduino
   - PIR sensor GND  →  GND on Arduino
   - PIR sensor OUT  →  Digital Pin 4 on Arduino
*/

const int PIR_SENSOR_OUTPUT_PIN = 4;  /* PIR sensor output pin connected to D4 */
int warm_up;

void setup() {
  pinMode(PIR_SENSOR_OUTPUT_PIN, INPUT);  /* Set PIR output pin as input */
  Serial.begin(9600);                     /* Begin serial communication */
  delay(2000);                            /* Power On Warm Up Delay for PIR */
}

void loop() {
  int sensor_output;
  sensor_output = digitalRead(PIR_SENSOR_OUTPUT_PIN);  /* Read PIR output */

  if(sensor_output == LOW) {               /* No motion detected */
    if(warm_up == 1) {
      Serial.print("Warming Up\n\n");
      warm_up = 0;
      delay(2000);
    }
    Serial.print("No object in sight\n\n");
    delay(1000);
  }
  else {                                   /* Motion detected */
    Serial.print("Object detected\n\n");    
    warm_up = 1;
    delay(1000);
  }  
}
